defer, panic, recover란? 🤔

defer = 함수 끝날 때 실행 예약! ⏰

panic = 심각한 에러 → 프로그램 중단! 💥

recover = panic을 잡아서 복구! 🛡️

세 가지가 함께 Go의 안전망을 구성! 🎯

다른 언어와 비교

→ Java: try-finally로 리소스 정리

→ Python: with문 + try-except

→ JS: try-catch-finally 구문

→ Go: defer + panic + recover! 🔥

핵심 특징

→ defer: LIFO 순서로 실행 (스택!) ✅

→ panic: 스택을 거슬러 올라가며 전파 🔧

→ recover: defer 안에서만 동작 🎯

→ 리소스 정리에 defer가 필수! 📊

Go 1분 정리 EP.18

defer 기본 사용법

func main() {
    fmt.Println("시작")
    defer fmt.Println("세 번째")
    defer fmt.Println("두 번째")
    defer fmt.Println("첫 번째")
    fmt.Println("끝")
}
// 출력:
// 시작
// 끝
// 첫 번째  ← LIFO (후입선출!)
// 두 번째
// 세 번째

// defer = 함수 종료 시 실행 예약
// 여러 개면 스택처럼 역순 실행!
// 인자는 defer 호출 시점에 평가됨

x := 10
defer fmt.Println(x) // 10 출력 (지금 평가!)
x = 20

⭐ defer는 LIFO! 나중에 등록한 게 먼저 실행!

Go 1분 정리 EP.18

defer로 리소스 정리

// 파일 열기 → defer로 닫기 보장!
func ReadFile(path string) ([]byte, error) {
    f, err := os.Open(path)
    if err != nil {
        return nil, err
    }
    defer f.Close() // 함수 끝나면 반드시 닫힘!

    return io.ReadAll(f)
}

// 뮤텍스 잠금 → defer로 해제!
var mu sync.Mutex

func SafeUpdate(data *Data) {
    mu.Lock()
    defer mu.Unlock() // 잠금 해제 보장!

    data.Count++
    data.UpdatedAt = time.Now()
    // 어떤 에러가 나도 Unlock 실행!
}

💡 열었으면 defer로 닫아라! Go의 철칙!

Go 1분 정리 EP.18

panic: 프로그램 중단

// panic = 복구 불가능한 심각한 에러!
func MustParseConfig(path string) Config {
    data, err := os.ReadFile(path)
    if err != nil {
        panic("설정 파일 없음: " + path)
    }
    var cfg Config
    if err := json.Unmarshal(data, &cfg);
        err != nil {
        panic("설정 파싱 실패: " + err.Error())
    }
    return cfg
}

// panic 발생 시:
// 1. 현재 함수 즉시 중단
// 2. defer들은 실행됨! ← 중요!
// 3. 호출 스택을 거슬러 올라감
// 4. recover 없으면 프로그램 종료

// ⚠️ 일반 에러는 error 반환 사용!
// panic은 정말 심각할 때만!

⚠️ panic은 최후의 수단! 보통은 error 반환!

Go 1분 정리 EP.18

recover: 패닉 복구

// recover = defer 안에서만 동작!
func SafeExecute(fn func()) (err error) {
    defer func() {
        if r := recover(); r != nil {
            err = fmt.Errorf("복구됨: %v", r)
        }
    }()

    fn() // panic이 발생해도 복구!
    return nil
}

// 사용 예시
err := SafeExecute(func() {
    panic("큰일이다!")
})
fmt.Println(err)
// 복구됨: 큰일이다!

// recover() 반환값:
// - panic 인자값 (interface{})
// - panic 없으면 nil

⭐ recover는 반드시 defer 함수 안에서!

실무: 안전한 고루틴

// 고루틴에서 panic → 전체 프로그램 종료!
// recover로 보호하는 패턴
func SafeGo(fn func()) {
    go func() {
        defer func() {
            if r := recover(); r != nil {
                log.Printf("고루틴 패닉: %v",
                    r)
                // 스택 트레이스 출력
                debug.PrintStack()
            }
        }()
        fn()
    }()
}

// HTTP 서버의 미들웨어 패턴
func Recovery(next http.Handler) http.Handler {
    return http.HandlerFunc(
        func(w http.ResponseWriter,
            r *http.Request) {
        defer func() {
            if err := recover(); err != nil {
                w.WriteHeader(500)
            }
        }()
        next.ServeHTTP(w, r)
    })
}

⭐ 고루틴 + recover = 서버 안정성의 핵심!

Go 1분 정리 EP.18

defer 고급 패턴

// 실행 시간 측정 패턴
func TrackTime(name string) func() {
    start := time.Now()
    return func() {
        fmt.Printf("%s: %v\n",
            name, time.Since(start))
    }
}

func Process() {
    defer TrackTime("Process")()
    // ... 작업 수행
    time.Sleep(2 * time.Second)
}
// Process: 2.001s

// 에러 래핑 패턴
func DoWork() (err error) {
    defer func() {
        if err != nil {
            err = fmt.Errorf(
                "DoWork 실패: %w", err)
        }
    }()
    return riskyOperation()
}

💡 defer + 클로저 = 강력한 실무 패턴!
