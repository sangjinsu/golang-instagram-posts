Go 1분 정리 EP.16

첫 번째 테스트 작성

// math.go
package calc

func Add(a, b int) int {
    return a + b
}

// math_test.go
package calc

import "testing"

func TestAdd(t *testing.T) {
    got := Add(2, 3)
    want := 5
    if got != want {
        t.Errorf("Add(2,3) = %d, want %d",
            got, want)
    }
}

⭐ go test 실행 → PASS 또는 FAIL 출력!

Go 1분 정리 EP.16

테이블 기반 테스트 (Table-Driven)

func TestAdd(t *testing.T) {
    tests := []struct {
        name string
        a, b int
        want int
    }{
        {"양수", 2, 3, 5},
        {"음수", -1, -2, -3},
        {"제로", 0, 0, 0},
        {"혼합", -1, 5, 4},
    }
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            got := Add(tt.a, tt.b)
            if got != tt.want {
                t.Errorf("got %d, want %d",
                    got, tt.want)
            }
        })
    }
}

⭐ 테이블 테스트 = Go 테스트의 황금 패턴!

테스트 실행 명령어

// 현재 패키지 테스트

Go 1분 정리 EP.16

에러 테스트와 헬퍼

func TestDivide(t *testing.T) {
    // 정상 케이스
    got, err := Divide(10, 2)
    if err != nil {
        t.Fatalf("에러 발생: %v", err)
    }
    if got != 5 {
        t.Errorf("got %d, want 5", got)
    }
    // 에러 케이스
    _, err = Divide(10, 0)
    if err == nil {
        t.Error("에러가 발생해야 함")
    }
}
// t.Error  = 실패 기록, 계속 실행
// t.Fatal  = 실패 기록, 즉시 중단
// t.Skip   = 테스트 건너뛰기

💡 Fatal = 즉시 중단, Error = 계속 실행!

Go 1분 정리 EP.16

벤치마크 테스트

// BenchmarkXxx(b *testing.B)
func BenchmarkAdd(b *testing.B) {
    // b.N = Go가 자동 결정!
    for i := 0; i < b.N; i++ {
        Add(100, 200)
    }
}
// $ go test -bench=.
// BenchmarkAdd-8  1000000000
//                 0.3171 ns/op

// 메모리 할당 확인
// $ go test -bench=. -benchmem
// 0 allocs/op ← 힙 할당 없음!

// benchstat으로 벤치마크 비교 가능
// $ go test -bench=. -count=5

⭐ -bench=. 으로 성능 측정, -benchmem 메모리!

Go 1분 정리 EP.16

실전: TestMain과 셋업

// TestMain = 테스트 전후 셋업/정리
func TestMain(m *testing.M) {
    // 전체 셋업
    fmt.Println("셋업 시작")
    db := setupTestDB()

    // 모든 테스트 실행
    code := m.Run()

    // 정리 (cleanup)
    db.Close()
    fmt.Println("정리 완료")
    os.Exit(code)
}
// t.Cleanup = 개별 테스트 정리
func TestUser(t *testing.T) {
    f := createTempFile(t)
    t.Cleanup(func() { os.Remove(f) })
}

⭐ TestMain = 전체 셋업, t.Cleanup = 개별 정리!
