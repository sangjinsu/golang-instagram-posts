동시성 패턴이란? 🤔

고루틴 + 채널을 조합하는 설계 패턴! 🧩

select = 여러 채널을 동시에 대기 📡

sync 패키지 = 공유 자원 안전하게 관리 🔒

실무에서 반드시 알아야 할 패턴들! 🎯

동시성 vs 병렬성

→ 동시성: 여러 작업을 번갈아 처리 🔄

→ 병렬성: 여러 작업을 동시에 처리 ⚡

→ Go: 동시성을 쉽게 → 병렬성은 자동!

→ 고루틴 = 동시성의 핵심 도구! 🚀

핵심 도구 3가지

→ select: 다중 채널 멀티플렉싱 ✅

→ sync.WaitGroup: 고루틴 완료 대기 🔧

→ sync.Mutex: 공유 자원 보호 🔒

→ sync.Once: 딱 한번만 실행 보장 🎯

Go 1분 정리 EP.21

select: 다중 채널 대기

func main() {
    ch1 := make(chan string)
    ch2 := make(chan string)

    go func() {
        time.Sleep(1 * time.Second)
        ch1 <- "채널 1 완료"
    }()
    go func() {
        time.Sleep(2 * time.Second)
        ch2 <- "채널 2 완료"
    }()

    // 먼저 도착하는 채널 처리!
    select {
    case msg := <-ch1:
        fmt.Println(msg)
    case msg := <-ch2:
        fmt.Println(msg)
    case <-time.After(3 * time.Second):
        fmt.Println("타임아웃!")
    }
    // 채널 1 완료 (1초 후 먼저 도착!)

⭐ select = switch의 채널 버전! 먼저 온 놈이 승!

WaitGroup: 고루틴 완료 대기

func main() {
    var wg sync.WaitGroup

    urls := []string{
        "https://api.example.com/users",
        "https://api.example.com/posts",
        "https://api.example.com/comments",
    }

    for _, url := range urls {
        wg.Add(1) // 카운터 +1
        go func(u string) {
            defer wg.Done() // 카운터 -1
            resp, _ := http.Get(u)
            defer resp.Body.Close()
            fmt.Printf("%s → %d\n",
                u, resp.StatusCode)
        }(url)
    }

    wg.Wait() // 카운터가 0이 될 때까지 대기!
    fmt.Println("모든 요청 완료!")
}

💡 Add(1) → Done() → Wait() = 3단계 패턴!

Mutex: 공유 자원 보호

type SafeCounter struct {
    mu sync.Mutex
    count map[string]int
}

func (c *SafeCounter) Inc(key string) {
    c.mu.Lock()         // 잠금!
    defer c.mu.Unlock() // 해제!
    c.count[key]++
}

func (c *SafeCounter) Get(key string) int {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.count[key]
}

func main() {
    c := SafeCounter{
        count: make(map[string]int)}
    for i := 0; i < 1000; i++ {
        go c.Inc("key")
    }
    time.Sleep(time.Second)
    fmt.Println(c.Get("key")) // 1000 ✅
}

⭐ Mutex 없이 map 동시 접근 = panic! 💥

Go 1분 정리 EP.21

Fan-Out / Fan-In 패턴

// Fan-Out: 하나의 입력 → 여러 워커
// Fan-In: 여러 워커 → 하나의 출력
func worker(id int,
    jobs <-chan int,
    results chan<- int) {
    for j := range jobs {
        fmt.Printf("워커%d: 작업%d 처리\n",
            id, j)
        time.Sleep(time.Second)
        results <- j * 2
    }
}

func main() {
    jobs := make(chan int, 100)
    results := make(chan int, 100)

    // Fan-Out: 3개 워커 실행
    for w := 1; w <= 3; w++ {
        go worker(w, jobs, results)
    }

    // 작업 전송
    for j := 1; j <= 9; j++ {
        jobs <- j
    }
    close(jobs)

    // Fan-In: 결과 수집
    for i := 0; i < 9; i++ {
        fmt.Println(<-results)
    }
}

⭐ 워커 풀 패턴 = Go 동시성의 꽃!

Go 1분 정리 EP.21

실무: errgroup 패턴

import "golang.org/x/sync/errgroup"

func FetchAll(ctx context.Context,
    urls []string) ([]Response, error) {
    g, ctx := errgroup.WithContext(ctx)
    results := make([]Response, len(urls))

    for i, url := range urls {
        g.Go(func() error {
            resp, err := FetchURL(ctx, url)
            if err != nil {
                return err // 에러 → 전체 취소!
            }
            results[i] = resp
            return nil
        })
    }

    // 모든 고루틴 완료 대기
    if err := g.Wait(); err != nil {
        return nil, err
    }
    return results, nil
}
// errgroup = WaitGroup + error + context
// 하나 실패 → 나머지 자동 취소! 🎯

⭐ errgroup = 실무 동시성의 표준 도구!

sync.Once와 RWMutex

// sync.Once = 딱 한 번만 실행!
var (
    instance *Database
    once     sync.Once
)

func GetDB() *Database {
    once.Do(func() {
        instance = ConnectDB()
        fmt.Println("DB 연결 완료")
    })
    return instance // 이후 호출은 캐시된 값
}

// sync.RWMutex = 읽기는 동시에, 쓰기만 잠금!
type Cache struct {
    mu   sync.RWMutex
    data map[string]string
}

func (c *Cache) Get(key string) string {
    c.mu.RLock()         // 읽기 잠금 (동시 OK)
    defer c.mu.RUnlock()
    return c.data[key]
}

func (c *Cache) Set(key, val string) {
    c.mu.Lock()          // 쓰기 잠금 (배타적)
    defer c.mu.Unlock()
    c.data[key] = val
}

💡 Once = 싱글톤, RWMutex = 읽기 많을 때!
