Context 기본: Background & TODO

import "context"

// 최상위 Context 2가지
ctx := context.Background()
// → 루트 context, 절대 취소 안 됨
// → main, 서버 시작점에서 사용

ctx2 := context.TODO()
// → "아직 뭘 쓸지 모르겠다"
// → 리팩토링 중 임시로 사용

// Context는 트리 구조!
//
//     Background()
//       ├── WithCancel()
//       │     └── WithTimeout()
//       └── WithValue()
//
// 부모 취소 → 자식도 모두 취소!
// 자식 취소 → 부모에 영향 없음!

⭐ Background() = 모든 context의 시작점!

WithCancel: 수동 취소

func main() {
    ctx, cancel := context.WithCancel(
        context.Background())

    go func(ctx context.Context) {
        for {
            select {
            case <-ctx.Done():
                fmt.Println("작업 취소됨!")
                return
            default:
                fmt.Println("작업 중...")
                time.Sleep(500 * time.Millisecond)
            }
        }
    }(ctx)

    time.Sleep(2 * time.Second)
    cancel() // 취소 신호 전송!
    time.Sleep(time.Second)
}
// 작업 중...
// 작업 중...
// 작업 중...
// 작업 취소됨!

💡 cancel() 호출 → ctx.Done() 채널 닫힘!

WithTimeout: 시간 초과 취소

func FetchData(ctx context.Context,
    url string) ([]byte, error) {
    // 3초 타임아웃 설정
    ctx, cancel := context.WithTimeout(
        ctx, 3*time.Second)
    defer cancel() // 항상 defer로 cancel!

    req, _ := http.NewRequestWithContext(
        ctx, "GET", url, nil)

    resp, err := http.DefaultClient.Do(req)
    if err != nil {
        return nil, err // 타임아웃 시 에러!
    }
    defer resp.Body.Close()

    return io.ReadAll(resp.Body)
}

// ctx.Err() 로 원인 확인
// context.DeadlineExceeded → 시간 초과
// context.Canceled → 수동 취소

⭐ WithTimeout = 가장 많이 쓰는 패턴!

WithValue: 값 전달

// 요청 범위 값 전달 (인증 정보 등)
type ctxKey string

const userKey ctxKey = "user"

func WithUser(ctx context.Context,
    user string) context.Context {
    return context.WithValue(
        ctx, userKey, user)
}

func GetUser(ctx context.Context) string {
    user, ok := ctx.Value(userKey).(string)
    if !ok {
        return "anonymous"
    }
    return user
}

ctx := WithUser(
    context.Background(), "gopher")
fmt.Println(GetUser(ctx)) // gopher

// ⚠️ 주의: 키는 비공개 타입 사용!
// string 키 → 충돌 위험! 💥

⚠️ WithValue는 요청 범위 데이터만! 남용 금지!

실무: HTTP 핸들러 + Context

func OrderHandler(
    w http.ResponseWriter,
    r *http.Request) {
    // 요청 context = 클라이언트 끊으면 취소!
    ctx := r.Context()

    // DB 쿼리에 context 전달
    order, err := db.QueryContext(ctx,
        "SELECT * FROM orders WHERE id=$1",
        r.URL.Query().Get("id"))
    if err != nil {
        if ctx.Err() != nil {
            log.Println("클라이언트 취소")
            return
        }
        http.Error(w, err.Error(), 500)
        return
    }
    json.NewEncoder(w).Encode(order)
}

// Context 전달 규칙:
// 1. 첫 번째 인자로 전달
// 2. 구조체에 저장하지 말 것!
// 3. nil context 전달 금지!

⭐ HTTP 요청 → DB → 외부API 모두 context 전달!

실무: 다중 고루틴 취소

func ProcessAll(ctx context.Context,
    items []Item) error {
    ctx, cancel := context.WithCancel(ctx)
    defer cancel()

    errCh := make(chan error, len(items))

    for _, item := range items {
        go func(it Item) {
            errCh <- Process(ctx, it)
        }(item)
    }

    for range items {
        if err := <-errCh; err != nil {
            cancel() // 하나 실패 → 전체 취소!
            return err
        }
    }
    return nil
}
// 하나의 cancel()로
// 수십 개 고루틴을 한번에 중단! 🎯

💡 cancel 하나로 모든 자식 고루틴 정리!
